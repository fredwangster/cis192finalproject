    function triggerSearch() {
        var txtSearchBox = document.getElementById('Search_String');
        var searchString = txtSearchBox.value;
        if (searchString.length > 0) {
            var f = document.forms['SearchForm'];
            if (!f) {
                f = document.SearchForm;
            }
            //Modified to handle multiple clicks of search button

            if (aPosSearchUrl > 0) {
                f.action = f.action.replace(getQuerystring('searchstring', f.action), encodeURI(searchString));
            }
            else {
                f.action = f.action + "&searchstring=" + encodeURI(searchString);
            }
            f.submit();
        }
    }
    function getQuerystring(key, url) {
        key = key.replace(/[\[]/, "\\\[").replace(/[\]]/, "\\\]");
        var regex = new RegExp("[\\?&]" + key + "=([^&#]*)");
        var qs = regex.exec(url);
        if (qs == null)
            return "";
        else
            return qs[1];
    }

                                document.getElementById(errorline).style.marginTop = '13px';
