		ticker.items.each(function() {
			$(this).show();
		});
		
		setTimeout("rotate_ticker()", ticker.delay);
		checkSize();
	}
	
	// Anto
	$(window).resize(checkSize);
	
	function rotate_ticker() {
		checkSize();
		
		ticker.current++;
		if (ticker.current >= ticker.items.length) {
		ticker.current = 0;
		}
		
		ticker.ul.find('li').eq(0).animate({
			marginLeft: '-325px'
		}, 750, "easeInQuart", function(){
			$(this).remove();
		});
		
		ticker.ul.append(ticker.items.eq(ticker.current).clone());
		
		setTimeout("rotate_ticker()", ticker.delay);
	}
	
	// Anto
	function checkSize()
	{
		ticker.max = parseInt(($(window).width() - ticker.marginSize) / ticker.itemSize);
		ticker.curItemCount = ticker.ul.find('li').size();
		
		if(ticker.curItemCount == ticker.max)
		{
			return;
		}
		else if(ticker.curItemCount < ticker.max)
		{
			ticker.current++;
			if (ticker.current >= ticker.items.length)
			{
				ticker.current = 0;
			}
			ticker.ul.append(ticker.items.eq(ticker.current).clone());
			//$('#log').append('<div>Added another element</div>');
		}
		else if (ticker.curItemCount > ticker.max)
		{
			ticker.current--;
			ticker.ul.find('li:last').remove();
			//$('#log').append('<div>Removed an element</div>');
		}
		checkSize();
	}
