function actTab( elm ) {
	var curElm = $(elm);
	var nextElm = $(curElm.id).nextSiblings();
	var preElm = $(curElm.id).previousSiblings();

	curElm.className = 'act';

	if ( preElm[0].className.match('div') ) preElm[0].addClassName("div leftact");
	if ( nextElm[0].className.match('div') ) nextElm[0].addClassName("div rightact");
	if ( preElm[0].className.match('first') ) preElm[0].addClassName("firstact");
	if ( nextElm[0].className.match('last') ) nextElm[0].addClassName("lastact");
}
function changeAct( elm, tab ) {
	var curElm = tab ? $(elm) : elm.parentNode;
	var nextElm = $(curElm.id).nextSiblings();
	var preElm = $(curElm.id).previousSiblings();

	curElm.className = 'act';

	if ( preElm[0].className.match('div') ) {
		if ( preElm[0].className.match('rightact') ) {
			preElm[0].removeClassName("rightact");
			preElm[0].addClassName("leftact2");
		} else {
			preElm[0].addClassName("leftact");
		}
	}
	if ( nextElm[0].className.match('div') ) {
		if ( nextElm[0].className.match('leftact') ) {
			nextElm[0].removeClassName("leftact");
			nextElm[0].addClassName("rightact2");
		} else {
			nextElm[0].addClassName("rightact");
		}
	}
	if ( preElm[0].className.match('first') ) preElm[0].addClassName("firstact");
	if ( nextElm[0].className.match('last') ) nextElm[0].addClassName("lastact");
}

function changeInact( elm ) {
	if ( elm.parentNode.id != curTab ) {
		var curElm = elm.parentNode;
		var nextElm = $(curElm.id).nextSiblings();
		var preElm = $(curElm.id).previousSiblings();
	
		curElm.className = '';
		if ( preElm[0].className.match('div') ) {
			if ( preElm[0].className.match('leftact2') ) {
				preElm[0].removeClassName("leftact2");
				preElm[0].addClassName("rightact");
			} else {
				preElm[0].removeClassName("leftact");
			}
		}
		if ( nextElm[0].className.match('div') ) {
			if ( nextElm[0].className.match('rightact2') ) {
				nextElm[0].removeClassName("rightact2");
				nextElm[0].addClassName("leftact");
			} else {
				nextElm[0].removeClassName("rightact");
			}
		}
		if ( preElm[0].className.match('first') ) {
			preElm[0].removeClassName("firstact");
			preElm[0].addClassName("first");
		}
		if ( nextElm[0].className.match('last') ) {
			nextElm[0].removeClassName("lastact");
			nextElm[0].addClassName("last");
		}
		
	}
}

function toggle( elm, manual ) {
	if ( manual ) {
		content_id = elm;
	} else {
		var content = elm.getElementsByTagName('div');
		var content_id = content[0].id;
	}
	
	if ( toggle_f != content_id ) {
		if ( toggle_f != "" ) { 
			$(toggle_f).parentNode.className = 'plus';
			Effect.SlideUp( toggle_f, { duration: 0.3 } );
		}
		Effect.SlideDown(content_id, { duration: 0.3 } );
		toggle_f = content_id;
        if ( manual ) {
			$(toggle_f).parentNode.className = 'minus';
        } else {
		    elm.className = 'minus';
        }
	}
}
