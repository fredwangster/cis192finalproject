InstantWinGame.prototype.getInstantWinResult = function() {
	
	
		toggleAnimationInterrupt(true);
	
	//quick check for old bot
	jackass247 = "&hack=false";	
	sUrl = "/winnerchk.jhtml?gameID=" + this.gameId + "&rand=" + Math.random() + jackass247;
	randomPrizeId = (Math.floor(Math.random() * 4) + 20) * -1;
	c = "game" + currentInstantWinGame;	
	IWGErrorTimer = setTimeout('forceInstantWinStop()',10000);
	//instantWinGamesData[currentInstantWinGame].Score --;
	//reloadSpinsLoadVar();
	var self = this;
	jQuery.ajax({
		url: sUrl,
		method: 'GET',
		dataType: 'xml',
		success:  function(xml) {
			clearTimeout(IWGErrorTimer);
			var responseXML = xml;   
			var status = responseXML.getElementsByTagName("status")[0].firstChild.data;
			var outcome = "LOSE";
			prizeId = 0;
			if (status == "OK") {
				var resultXML = responseXML.getElementsByTagName("result")[0];
				var bWinner = (resultXML.getElementsByTagName("status")[0].firstChild.data == "Win") ? true : false;
				var spins = resultXML.getElementsByTagName("plays-remaining")[0].firstChild.data;
				instantWinGamesData[currentInstantWinGame].Score = instantWinGamesData[currentInstantWinGame].Score - 1;
				reloadSpinsLoadVar();
				var prizeId = resultXML.getElementsByTagName("prize-id")[0].firstChild.data;
				if (bWinner) {				
					var claimCode = resultXML.getElementsByTagName("claim-code")[0].firstChild.data;
					outcome = "WIN";
					self.claimCode = claimCode;
					self.prizeId = prizeId;
					self.setInstantWinResult(outcome, spins, prizeId, claimCode);
				} else {
					if (prizeId != 0) {
						if (resultXML.getElementsByTagName("prize-type")[0]) {
							var prizeType = resultXML.getElementsByTagName("prize-type")[0].firstChild.data;
						}
						else {
							var prizeType = 'nothing';
						}					
						if (prizeType == 'coins'){
							//Player gets coins
							lastWinType = "coins";
							var prizeCoins = resultXML.getElementsByTagName("coins")[0].firstChild.data;
							NumberOf['TotalMedallions'] = prizeCoins;
							outcome = "WIN";
							claimCode = "consolation";
							
						}
						else if (prizeType == 'spins'){
							//Player gets spins
							lastWinType = "spins";
							//awardedSpins(instantWinGamesData[currentInstantWinGame].Score);
							oldSpins = instantWinGamesData[currentInstantWinGame].Score;
							outcome = "WIN";
							claimCode = "consolation";
							////alert('spins: ' + spins + ' oldspins: ' + oldSpins );	
						}
						else {
							//Player really loses
							lastWinType = "nothing";
						}					
					}
					else {
						////alert('loss');
						prizeId = randomPrizeId;	
						spins = instantWinGamesData[currentInstantWinGame].Score;
					}		
					//window.setTimeout(function() {  self.setInstantWinResult(outcome); }, 2 * 1000);
					self.setInstantWinResult(outcome, instantWinGamesData[currentInstantWinGame].Score, prizeId, claimCode);
					//awardedSpins(instantWinGamesData[currentInstantWinGame].Score);
					instantWinGamesData[currentInstantWinGame].Score = spins;
					reloadSpinsLoadVar();				
				}			
			} else {
				outcome = "ERROR";
				error = responseXML.getElementsByTagName("error")[0].firstChild.data;
				if (error == "STATUS_AT_MAX") {
					self.setInstantWinResult(outcome, 0, randomPrizeId);
				} else {
					self.setInstantWinResult(outcome, 1, randomPrizeId);
				}
			}
		},
		error: function() {
			clearTimeout(IWGErrorTimer); 
			outcome = "ERROR"; 
			self.setInstantWinResult(outcome, instantWinGamesData[currentInstantWinGame].Score, randomPrizeId); 
		},
		complete:function(jqXHR, textStatus){
			//nothing	
		}
	});

	//clearInterval(changeIWGTimer);
}
