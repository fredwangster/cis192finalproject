function saxotech2tabletUrl(type,catId,pageId){
	var url;
	switch(type){
		case 'article':
			// New article ID.
			pageId = GEL.thepage.pageinfo.date.isopuddate.substr(0,4).concat(GEL.thepage.pageinfo.key);
			// Get last 9 characters.
			pageId = pageId.substr(pageId.length-9);
			// Add year back to the beginning.
			pageId = GEL.thepage.pageinfo.date.isopuddate.substr(0,4).concat(pageId);
			url='http://tablet.indystar.com/index.php/main/article/'+catId.toLowerCase()+'/'+pageId.toLowerCase();
		break;
		case 'gallery':
			pageId = GEL.thepage.pageinfo.sitecode + '|' + GEL.thepage.pageinfo.date.isopuddate + '|' + catId + '|' + pageId + '|' + 'PH';
			url='http://tablet.indystar.com/index.php/main/gallery/'+pageId;
		break;
		case 'section':
			url='http://tablet.indystar.com/index.php/main/category/'+catId.toLowerCase();
		break;
		case 'frontpage':
		default:
			url='http://tablet.indystar.com';
		break;
	}
	return url;
}
