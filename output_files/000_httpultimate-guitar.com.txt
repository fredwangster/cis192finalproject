<script type="text/javascript">
var autoscroll = 700;
var slide = 300;
var interval = -1;
var current_screen_id = 0;
var img_width = 774;
var animate_run = 0;
var animate_side = 0;

function start_slideshow()
{
	animate_run = 0;
    //interval = setTimeout(slideshow, autoscroll);

}

function toggle_slide(delta)
{
	animate_side = 0;
	animate_run = 1;
	$('#slides').animate({
		marginLeft: delta
	}, slide, start_slideshow);

	$('#slidedots span').removeClass('ison');
	$('#slidedots span:eq(' + current_screen_id + ')').addClass('ison');
}
function toggle_slide2()
{ 	
	/*current_screen_id=5;
	$('#slidedots span').removeClass('ison');
	$('#slidedots span:eq(' + current_screen_id + ')').addClass('ison');
	*/
    next_screen_id = parseInt($.cookie('next_screen_id'));
	next_screen_id = next_screen_id ? next_screen_id : 0;
	next_screen_id++;
	next_screen = $('#slides span:eq(' + next_screen_id + ')');
	if ($('#slides span').index($(next_screen)) == -1)
	{
		next_screen_id = 0;
	}
	
	$.cookie('next_screen_id', next_screen_id, { expires: 60, path: '/' });
}

function slideshow()
{
    current_screen_id = Math.abs(parseInt($('#slides').css('margin-left'), 10) / img_width);
	if (!animate_run && current_screen_id == Math.round(current_screen_id))
	{
		next_screen_id = current_screen_id + (animate_side ? -1 : 1);
		next_screen = $('#slides span:eq(' + next_screen_id + ')');
		if ($('#slides span').index($(next_screen)) != -1)
		{
			current_screen_id = next_screen_id;
			toggle_slide((animate_side ? '+' : '-') + '=' + img_width);
		}
		else
		{
			current_screen_id = animate_side ? $('#slides span').length - 1 : 0;
			toggle_slide(-current_screen_id * img_width + 'px');
		}
	}
}
function slideshow2()
{
    current_screen_id = Math.abs(parseInt($('#slides').css('margin-left'), 10) / img_width);
	if (!animate_run && current_screen_id == Math.round(current_screen_id))
	{
		next_screen_id = current_screen_id + (animate_side ? -1 : 1);
		next_screen = $('#slides span:eq(' + next_screen_id + ')');
		if ($('#slides span').index($(next_screen)) != -1)
		{
			current_screen_id = next_screen_id;
			toggle_slide2((animate_side ? '+' : '-') + '=' + img_width);
		}
		else
		{
			current_screen_id = animate_side ? $('#slides span').length - 1 : 0;
			toggle_slide2(-current_screen_id * img_width + 'px');
		}
	}
}

$(document).ready(function(){
	$(".seer").hover(function() {clearTimeout(interval)}, function() {interval = setTimeout(slideshow, autoscroll);});
	$('#slides span').each(function(index) {
		$('#slidedots').append('<span' + (!index ? ' class="ison"' : '') + '></span> ');
	});
    start_slideshow();
    $('#slidedots span').click(function() {
		if (!animate_run)
		{
			dot_id = $('#slidedots span').index($(this));
			if (current_screen_id != dot_id)
			{
				delta = (current_screen_id < dot_id ? '-' : '+') + '=' + Math.abs(img_width * (current_screen_id - dot_id));
				clearTimeout(interval);
				current_screen_id = dot_id;
				toggle_slide(delta);
			}
		}

		return false;
    });

    $('#right, #left').click(function() {
		if (!animate_run)
		{
			animate_side = $(this).attr('id') == 'left';
			clearTimeout(interval);
			slideshow();
		}

        return false;
    });

});
