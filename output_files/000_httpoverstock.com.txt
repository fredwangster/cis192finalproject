os.AutoComplete = function( options ) {
	var opt = options || {},
		$dropdown, 
		$searchInput = jQuery('#search-input'),
		$searchBox = jQuery('#search-bar-container'),
		searchTerm = '',
		chosen = "",
		filter = opt.filter,
		$chosen = null;
		
	return {
		init: function() {
			this.setUp();
			this.bindEventListeners();
		},
		bindEventListeners: function() {
			var _this = this;
			$searchInput.keyup(function(e) {
				searchTerm = this.value; //get value of input
				
				if(searchTerm && searchTerm.length > 0 && e.keyCode != 38 && e.keyCode != 40) {
					chosen = -1;
					_this.makeRequest(searchTerm);
				}
				else if( searchTerm === '' ){
					_this.clearUI();
				}
				else {
				 	_this.handleKeyEvents(e);
				 	$searchInput.val($dropdown.find('li.selected').text());
				}
		
				
			});
			
			$dropdown.bind("mouseover", function(e){
				
				var attr = e.target.getAttribute("data-index");
				
				if (attr) {
					chosen = parseInt(attr);
				
				}
			
				if ($chosen) {
					$dropdown.removeClass('chosen');
					$chosen.removeClass('selected');
					$chosen = null;
				}
			});
						
			$searchInput.blur(function() {
				_this.clearUI();
			});
		},
		buildUL: function(searchTermArray) {
			var l = Math.min(searchTermArray.length, 10),
				html = '<ul>',
				end = '</ul>';
			for (var i = 0; i < l; i++) {
				html += this.buildLi(searchTermArray[i], i);
			}
			return html += end;
		},
		
		handleKeyEvents: function(e) {
		
			//down	
			if (e.keyCode == 40) {
				if (chosen === "") {
					chosen = 0;
				} else if ((chosen + 1) < $('#os-autocomplete-container li').length) {
					chosen++;
				}
				$dropdown.addClass('chosen');
				jQuery('#os-autocomplete-container li').removeClass('selected');
				$chosen = jQuery('#os-autocomplete-container li:eq(' + chosen + ')');
				$chosen.addClass('selected');
				return false;
							
			}
			//up
			if (e.keyCode == 38) {
				if (chosen === "") {
					chosen = 0;
				} else if (chosen > 0) {
					chosen--;
				}
				$dropdown.addClass('chosen');
				jQuery('#os-autocomplete-container li').removeClass('selected');
				$chosen = jQuery('#os-autocomplete-container li:eq(' + chosen + ')');
				$chosen.addClass('selected');
				return false;
								
			}
			
		},

		
		buildLi: function(results, index) {
			var sTerm = new RegExp(searchTerm, 'gi');
			var matchResults = results.match(sTerm);
			
			if (matchResults != null) {
				var highlightTerm = "<span class='font-weight-normal'>" + matchResults[0] + "</span>";
				var replaceTerm = results.replace(matchResults[0], highlightTerm);
				var encodedResult = encodeURI( results );
				
				return '<li data-index="' + index + '"><a data-index="' + index + '" href="http://www.overstock.com/search?keywords=' + encodedResult + '&SearchType=Header">' + replaceTerm + '</a></li>';
			}
			
			else { 
				return ''; 
				}
			
		},
		
		handleSuccess: function(data) {
			var ul = this.buildUL(data);
			this.updateUi(ul);
		},
		
		makeRequest: function(searchTerm) {
			var _this = this,
				encodedTerm = encodeURI(searchTerm);
			jQuery.ajax({
				url: 'http://content.atomz.com/autocomplete/sp10/04/aa/0a/?term=' + encodedTerm + '&query=' + encodedTerm,
				dataType: 'jsonp',
				success: function(data) {
					var result = ( filter === true ) ? _this.filterResults( data, searchTerm ) : data; //["red","red shoes","red shirts"]
					
					_this.handleSuccess(result);
				}
				//error: function() {}
			});
		},
		
		filterResults: function( results, searchTerm ){
			
			var filteredResults = [],
				regEx = new RegExp( '^' + searchTerm, 'i'); //filters results that only start with searchTerm
			
			if( results && results.length > 0 ){
				
				for(var i=0, l=results.length; i<l; i++){
					if( regEx.test(results[i]) ){
						filteredResults.push( results[i] );
					}
					else {
						
					}
				}
				
			}
			//return results;
			return filteredResults;
			
		},
		
		setUp: function() {
			$dropdown = jQuery('<div id="os-autocomplete-container"></div>');
			$searchBox.after($dropdown);
		},
		
		updateUi: function(markup) {
			if ($dropdown.css('display') == 'none') {
				$dropdown.show();
			} //$dropdown.show().append( markup );
			$dropdown[0].innerHTML = markup;
		},
		clearUI: function() {
			setTimeout(function() {
				$dropdown.hide();
			}, 250);
		}
	}.init();

			os.TriggerEvent = function(event,delay,iterations){
				
				var e = event,
					d = delay || 0,
					i = iterations || 1,
					to,
					isPaused = false,
					isDisabled = false;
				
				return {
					
					init: function(){
						if(e){
							
							this.trigger(e,d);
						}
					},
					
					trigger: function(){
						_this = this;
						
						if(i>0 && !isPaused && !isDisabled){
							to = window.setTimeout(function(){
									if(e()){
										_this.trigger();
									}
									else {
										window.clearTimeout(to);
										_this.disable();
									}
							},d);
						}
						i--;
						
					},
					
					pause: function(){
						
						window.clearTimeout(to);
						isPaused = true;
						i++;
					},
					
					play: function(){
						
						isPaused = false;
						this.trigger();
					},
					
					disable: function(){
						isDisabled = true;
					},
					
					enable: function(){
						//not implemented
						isDisabled = false;
					}
					
				}
				
			}
